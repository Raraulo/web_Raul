3. Reemplazo de valores nulos por la media

In [11]:
from pyspark.sql.functions import mean, col
from pyspark.sql.types import DoubleType, FloatType, IntegerType, LongType

# Identificar columnas numéricas para el cálculo de la media
numeric_cols = [f.name for f in df_spark.schema.fields if isinstance(f.dataType, (DoubleType, FloatType, IntegerType, LongType))]

# Verificar si hay columnas numéricas para evitar errores
if not numeric_cols:
    print("No se encontraron columnas numéricas para el reemplazo de nulos. El DataFrame 'df_spark_filled' será el mismo que 'df_spark'.")
    df_spark_filled = df_spark # No hay columnas numéricas que procesar, mantener el original
else:
    # Calcular la media para cada columna numérica
    # Usamos .collect()[0].asDict() para obtener un diccionario de medias
    # Aseguramos que el DataFrame no esté vacío antes de intentar calcular la media
    if df_spark.count() > 0:
        means = df_spark.agg(*[mean(c).alias(c) for c in numeric_cols]).collect()[0].asDict()
    else:
        # Si el DataFrame está vacío, no podemos calcular medias, podemos inicializar con 0 o dejar nulos.
        # Para este caso, asumimos que el DataFrame no estará vacío después de la carga.
        print("Advertencia: El DataFrame está vacío, no se pueden calcular las medias para el reemplazo.")
        means = {}

    # Reemplazar valores nulos con la media en las columnas numéricas
    df_spark_filled = df_spark.na.fill(means)

    print("Valores nulos reemplazados por la media en columnas numéricas.")

print("\nConteo de nulos después del reemplazo por columna:")
# Contar y mostrar los nulos en cada columna numérica después del reemplazo
for c in numeric_cols:
    null_count = df_spark_filled.filter(col(c).isNull()).count()
    print(f"  Columna '{c}': {null_count} nulos")

print("\nPrimeras 5 filas del DataFrame después del reemplazo:")
df_spark_filled.show(5)

import matplotlib.pyplot as plt
import seaborn as sns

# Muestrear el 25% de los datos y convertir a Pandas DataFrame para visualización
sample_df_pandas = df_spark_filled.sample(fraction=0.25, seed=42).toPandas()

plt.figure(figsize=(12, 6))

# Boxplot para Voltaje
plt.subplot(1, 2, 1) # 1 fila, 2 columnas, primer gráfico
sns.boxplot(y=sample_df_pandas['voltage'])
plt.title('Boxplot de Voltaje (Muestra del 25%)')
plt.ylabel('Voltaje')
plt.grid(axis='y', linestyle='--', alpha=0.7)

# Boxplot para Temperatura
plt.subplot(1, 2, 2) # 1 fila, 2 columnas, segundo gráfico
sns.boxplot(y=sample_df_pandas['temperature'])
plt.title('Boxplot de Temperatura (Muestra del 25%)')
plt.ylabel('Temperatura')
plt.grid(axis='y', linestyle='--', alpha=0.7)

plt.tight_layout()
plt.show()

# Explicación de los gráficos

print("Los boxplots muestran la distribución de 'voltage' y 'temperature' para una muestra del 25% de los datos.")
print("La caja representa el rango intercuartílico (IQR), con la línea central indicando la mediana.")
print("Los 'bigotes' extienden a los valores más extremos dentro de 1.5 veces el IQR de la caja.")

Por ejemplo, una caja pequeña indica baja variabilidad, mientras que muchos outliers pueden señalar anomalías en el sensor o el proceso.")
